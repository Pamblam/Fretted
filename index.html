Vertical Labelled Dm7/E Guitar Chord Chart

// Dm7/E Guitar Chord Chart
Fretted.Guitar().makeChord([
	new Fretted.String().setFret(5).setNote("A"),
	new Fretted.String().setFret(6).setNote("F"),
	new Fretted.String().setFret(5).setNote("C"),
	new Fretted.String().setFret(7).setNote("A"),
	new Fretted.String().setFret(7).setNote("B"),
	new Fretted.String().setNote("E"),
]).render(Fretted.IMAGE).then(img=>{
	document.getElementById("chordchart1").appendChild(img);
});

Horizontal Open C Guitar Chord Chart

// Horizontal Open C Guitar Chord Chart
Fretted.Guitar().orientation(Fretted.HORIZONTAL).makeChord([
	new Fretted.String(),
	new Fretted.String().setFret(2),
	new Fretted.String(),
	new Fretted.String().setFret(3),
	new Fretted.String().setFret(4),
	new Fretted.String().mute(),
]).render(Fretted.IMAGE).then(img=>{
	document.getElementById("chordchart2").appendChild(img);
});

B Ukulele Chord Chart W/ Highlighted root notes

// B Ukulele Chord Chart W/ Highlighted root notes
var blue = "#4286f4", yellow = "#ffe95b", green = "#5bffb7";
Fretted.Ukulele().makeChord([
	new Fretted.String().setFret(2).setNote("B").setBGColor(blue).setTextColor(yellow),
	new Fretted.String().setFret(2).setNote("G").setBGColor(green),
	new Fretted.String().setFret(3).setNote("D#").setBGColor(green),
	new Fretted.String().setFret(4).setNote("B").setBGColor(blue).setTextColor(yellow),
]).render(Fretted.IMAGE).then(img=>{
	document.getElementById("chordchart3").appendChild(img);
});

Guitar Scale (C Maj)

// Guitar Scale (C Maj)
var blue = "#4286f4", yellow = "#ffe95b", green = "#5bffb7";
var c_maj_scale = new Fretted.Scale().setNotes(['C','D','E','F','G','A','B']).setRootNote("C").setNoteStyle(green).setRootNoteStyle(blue, yellow);
Fretted.Guitar().makeScale(c_maj_scale).render(Fretted.IMAGE).then(img=>{
	document.getElementById("scale1").appendChild(img);
});

Vertical C Maj Scale on Mandolin

// Vertical C Maj Scale on Mandolin
Fretted.makeInstrument({
	name: 'Mandolin',
	frets: 20,
	strings: 4,
	doubleFretMarkers: [12],
	singleFretMarkers: [3,5,7,10,15],
	tuning: ["E","A","D","G"]
});
var blue = "#4286f4", yellow = "#ffe95b", green = "#5bffb7";
var c_maj_scale = new Fretted.Scale().setNotes(['C','D','E','F','G','A','B']).setRootNote("C").setNoteStyle(green).setRootNoteStyle(blue, yellow);
Fretted.Mandolin().orientation(Fretted.VERTICAL).makeScale(c_maj_scale).render(Fretted.IMAGE).then(img=>{
	document.getElementById("scale2").appendChild(img);
});
